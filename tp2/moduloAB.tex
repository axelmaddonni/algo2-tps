\begin{Interfaz}
  
  %\textbf{usa}: \tadNombre{}.
  
  \textbf{se explica con}: \tadNombre{AB($\sigma$)}.
  
  \textbf{g\'eneros}: \TipoVariable{ab($\sigma$)}.

  \Titulos{Operaciones}

  \InterfazFuncion{nil?}{\In{a}{ab($\sigma$)}}{Bool}
  {res $\igobs$ nil?(a)}
  []
  [Indica si el arbol est\'a vac\'io.]

  \InterfazFuncion{raiz}{\In{a}{ab($\sigma$)}}{$\sigma$}
  [$\neg$nil?(a)]
  {res $\igobs$ raiz(a)}
  []
  [Devulve el valor de la raiz del arbol.]
  [alias?]
  
  \InterfazFuncion{izq}{\In{a}{ab($\sigma$}}{ab($\sigma$)}
  [$\neg$nil?(a)]  
  {res $\igobs$ vacio()}
  []
  [Devulve un puntero apuntando al hijo izquierdo.]
  
  \InterfazFuncion{der}{\In{a}{ab($\sigma$}}{ab($\sigma$)}
  [$\neg$nil?(a)]  
  {res $\igobs$ vacio()}
  []
  [Devuelve un puntero apuntando al hijo derecho.]
  
  \InterfazFuncion{nil}{}{ab($\sigma$)}
  []
  {res $\igobs$ nil()}
  []
  [Devuelve un \'arbol binario vacio.]
  
  \InterfazFuncion{bin}{\In{i}{ab($\sigma$)}, \In{a}{$\sigma$}, \In{d}{ab($\sigma$)}}{ab($\sigma$)}
  []
  {res $\igobs$ bin(i,a,d)}
  []
  [Crea un \'arbol binario con raiz a, hijo derecho d e hijo izquierdo i.]

\end{Interfaz}

\begin{Algoritmos}

\begin{algorithm}
\caption{Implementaci\'on de nil?}
\begin{algorithmic}[0]
\Function{iNil?}{in a: $arbol_{AB}$}{$\rightarrow$ res: bool}
	\State res $\gets$ (a.izq == NULO $\wedge$ a.der == NULO)
\EndFunction
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Implementaci\'on de raiz}
\begin{algorithmic}[0]
\Function{iRaiz}{in a: $arbol_{AB}$}{$\rightarrow$ res: $\sigma$}
	\State res $\gets$ a.raiz
\EndFunction
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Implementaci\'on de izq}
\begin{algorithmic}[0]
\Function{iIzq}{in a: $arbol_{AB}$}{$\rightarrow$ res: $arbol_{AB}$}
	\State res $\gets$ a.izq
\EndFunction
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Implementaci\'on de der}
\begin{algorithmic}[0]
\Function{iDer}{in a: $arbol_{AB}$}{$\rightarrow$ res: $arbol_{AB}$}
	\State res $\gets$ a.der
\EndFunction
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Implementaci\'on de nil}
\begin{algorithmic}[0]
\Function{iNil}{}{$\rightarrow$ res: $arbol_{AB}$}
	\State res $\gets$ <NULO,NULO,NULO>
\EndFunction
\end{algorithmic}
\end{algorithm}

\begin{algorithm}
\caption{Implementaci\'on de bin}
\begin{algorithmic}[0]
\Function{iIzq}{in i: $arbol_{AB}$, in a: $\sigma$, in d: $arbol_{AB}$}{$\rightarrow$ res: $arbol_{AB}$}
	\State res $\gets$ nil()
	\State res.izq $\gets$ i
	\State res.der $\gets$ d
	\State res.raiz $\gets$ a
\EndFunction
\end{algorithmic}
\end{algorithm}

\end{Algoritmos}
